148  private void GetLookupSizes(SakuraiArchiveHeader* hdr)
149  {
150  //Read lookup offsets first and use them to get entry sizes at each offset.
151  bint* lookup = hdr->LookupEntries;
152  //First add each offset to the dictionary with size of 0.
153  //The dictionary will sort the offsets automatically, in case they aren't already.
154  for (int i = 0; i < hdr->_lookupEntryCount; i++)
155  {
156  int w = *(bint*) Address(lookup[i]);
157  if (!_lookupSizes.ContainsKey(w))
158  {
159  _lookupSizes.Add(w, 0);
160  }
161  }
162 
163  //Now go through each offset and calculate the size with the offset difference.
164  int prev = 0;
165  bool first = true;
166  int[] t = _lookupSizes.Keys.ToArray();
167  for (int i = 0; i < t.Length; i++)
168  {
169  int off = t[i];
170  if (first)
171  {
172  first = false;
173  }
174  else
175  {
176  _lookupSizes[prev] = off - prev;
177  }
178 
179  prev = off;
180  }
181 
182  //The last entry in the moveset file goes right up to the lookup offsets.
183  _lookupSizes[prev] = Offset(lookup) - prev;

234  private static void AddToTrie(SYMBMaskEntry* trie, SYMBMaskHeader* head, int id, SYMBHeader* table)
235  {
236  //If list is empty add the node and quit
237  if (head->_numEntries == 0)
238  {
239  trie[head->_numEntries++] = new SYMBMaskEntry(1, -1, -1, -1, id, 0);
240  return;
241  }
242 
243  string value = table->GetStringEntry(id);
244 
245  //String.IsNullOrEmpty(value)
246  if ((value ?? "") == "")
247  {
248  throw new ArgumentException("String is null or whitespace");
249  }
250 
251  SYMBMaskEntry search = trie[head->_rootId];
252  List<int> path = new List<int>
253  {
254  head->_rootId
255  };
256 
257  //Find the string that matches the current string in the trie. Needs to be done in order to determine where the important bit is in the string
258  while (search._flags == 0)
259  {
260  //Assume that strings are treated as having an infinite number of null chars following them
261  if (search._bit / 8 >= value.Length)
262  {
263  path.Add(search._leftId);
264  search = trie[search._leftId];
265  continue;
266  }
267 
268  // _leftId corresponds to bit=0, _rightId corresponds to bit=1
269  if (CheckBit(value, search._bit))
270  {
271  path.Add(search._rightId);
272  search = trie[search._rightId];
273  }
274  else
275  {
276  path.Add(search._leftId);
277  search = trie[search._leftId];
278  }
279  }
280 
281  string searchVal = table->GetStringEntry(search._stringId);
282 
283  //Can't add duplicate strings
284  if (searchVal == value)
285  {
286  throw new ArgumentException("Duplicate string");
287  }
288 
289  bool mismatch = false;
290  int minLength = Math.Min(searchVal.Length, value.Length);
291  short bit = 0;
292 
293  //Locate mismatching character between the two strings
294  for (short i = 0; i < minLength; i++)
295  {
296  if (value[i] != searchVal[i])
297  {
298  mismatch = true;
299  bit = (short) (8 * i);
300  break;
301  }
302  }
303 
304  bool right;
305 
306  //If a char was different one string does not contain the other
307  if (mismatch)
308  {
309  //Find where the bits differed
310  int cmpint = value[bit / 8] ^ searchVal[bit / 8];
311  bit += clz8[cmpint];
312 
313  //If the bit is 1 the string being added takes the left fork
314  right = CheckBit(value, bit);
315 
316  if (head->_numEntries == 1)
317  {
318  trie[1] = new SYMBMaskEntry(1, -1, -1, -1, id, 1);
319  trie[2] = new SYMBMaskEntry(0, bit, right ? 0 : 1, right ? 1 : 0, -1, -1);
320  head->_numEntries = 3;
321  head->_rootId = 2;
322  return;
323  }
324 
325  //If the mismatch bit is lower than the first mismatch bit the new branch will be the root of the tree
326  if (bit < trie[path[0]]._bit)
327  {
328  trie[head->_numEntries++] = new SYMBMaskEntry(1, -1, -1, -1, id, head->_numEntries / 2);
329 
330  if (right)
331  {
332  trie[head->_numEntries++] = new SYMBMaskEntry(0, bit, path[0], head->_numEntries - 2, -1, -1);
333  }
334  else
335  {
336  trie[head->_numEntries++] = new SYMBMaskEntry(0, bit, head->_numEntries - 2, path[0], -1, -1);
337  }
338 
339  head->_rootId = head->_numEntries - 1;
340  return;
341  }
342 
343  //Locate where the branch needs to be inserted
344  for (int i = 1; i < path.Count; i++)
345  {
346  if (trie[path[i]]._bit > bit || trie[path[i]]._flags == 1)
347  {
348  //Add leaf
349  trie[head->_numEntries++] = new SYMBMaskEntry(1, -1, -1, -1, id, head->_numEntries / 2);
350 
351  //Remap previous branch to point to new branch
352  if (trie[path[i - 1]]._leftId == path[i])
353  {
354  trie[path[i - 1]]._leftId = head->_numEntries;
355  }
356  else
357  {
358  trie[path[i - 1]]._rightId = head->_numEntries;
359  }
360 
361  //Create new branch
362  if (right)
363  {
364  trie[head->_numEntries++] =
365  new SYMBMaskEntry(0, bit, path[i], head->_numEntries - 2, -1, -1);
366  }
367  else
368  {
369  trie[head->_numEntries++] =
370  new SYMBMaskEntry(0, bit, head->_numEntries - 2, path[i], -1, -1);
371  }
372 
373  return;
374  }
375  }
376 
377  //This should never happen
378  throw new Exception("Error building tree, unexpected structure");
379  }
380 
381  //Since mismatch is false, one string is a substring of the other
382 
383  //The longer string is the one that takes the left branch
384  right = value.Length > searchVal.Length;
385  bit = (short) (minLength * 8);
386 
387  if (right)
388  {
389  //Find the first bit after the substring that's 1. Will always occur in the first 8 bits because 0x00 denotes string termination and thus isn't in value
390  bit += clz8[value[bit / 8]];
391 
392  //If path.Count == 1 the only value is a leaf
393  if (path.Count == 1)
394  {
395  trie[1] = new SYMBMaskEntry(1, -1, -1, -1, id, 1);
396  trie[2] = new SYMBMaskEntry(0, bit, 0, 1, -1, -1);
397  head->_numEntries = 3;
398  head->_rootId = 2;
399  return;
400  }
401 
402  //Update old branch, insert new branch and node, and quit. trie[path[path.Count-2]] is the last branch that was a comparison.
403  trie[head->_numEntries++] = new SYMBMaskEntry(1, -1, -1, -1, id, head->_numEntries / 2);
404 
405  int trace = path.Count - 2;
406 
407  if (trie[path[trace]]._leftId == path[trace + 1])
408  {
409  //Handling an extremely specific and annoying edge case
410  while (trie[path[trace]]._bit > bit)
411  {
412  trace--;
413  if (trace < 0)
414  {
415  //This node is actually the root of the tree
416  trie[head->_numEntries++] =
417  new SYMBMaskEntry(0, bit, path[0], head->_numEntries - 2, -1, -1);
418  head->_rootId = head->_numEntries - 2;
419  return;
420  }
421  }
422 
423  trie[path[trace]]._leftId = head->_numEntries;
424  }
425  else
426  {
427  trie[path[trace]]._rightId = head->_numEntries;
428  }
429 
430  trie[head->_numEntries++] = new SYMBMaskEntry(0, bit, path[trace + 1], head->_numEntries - 2, -1, -1);
431  return;
432  }
433 
434  //Find first bit comparison that happens after the substring ends
435  int index;
436  for (index = 0; trie[path[index]]._flags == 0 && trie[path[index]]._bit <= bit; index++)
437  {
438  }
439 
440  //Find the first bit that's 1 and isn't already used in the trie
441  int cmpVal = searchVal[bit / 8];
442  byte clzVal;
443  bool test = trie[path[index]]._flags == 0;
444  while (true)
445  {
446  clzVal = clz8[cmpVal];
447  if (clzVal == 8)
448  {
449  bit += 8;
450  cmpVal = searchVal[bit / 8];
451  continue;
452  }
453 
454  if (test && trie[path[index]]._bit <= bit + clzVal)
455  {
456  if (trie[path[index]]._bit == bit + clzVal)
457  {
458  cmpVal ^= (1 << 7) >> clzVal;
459  }
460 
461  test = trie[path[++index]]._flags == 0;
462  continue;
463  }
464 
465  bit += clzVal;
466  break;
467  }
468 
469  //If the trie is a single leaf the new branch is the root of the trie
470  if (head->_numEntries == 1)
471  {
472  trie[1] = new SYMBMaskEntry(1, -1, -1, -1, id, 1);
473  trie[2] = new SYMBMaskEntry(0, bit, 1, 0, -1, -1);
474  head->_numEntries = 3;
475  head->_rootId = 2;
476  return;
477  }
478 
479  //Update old branch, insert new branch and node, and quit
480  trie[head->_numEntries++] = new SYMBMaskEntry(1, -1, -1, -1, id, head->_numEntries / 2);
481 
482  if (trie[path[index - 1]]._leftId == path[index])
483  {
484  trie[path[index - 1]]._leftId = head->_numEntries;
485  }
486  else
487  {
488  trie[path[index - 1]]._rightId = head->_numEntries;
489  }
490 
491  trie[head->_numEntries++] = new SYMBMaskEntry(0, bit, head->_numEntries - 2, path[index], -1, -1);
492 
493  return;
494  }
495  }

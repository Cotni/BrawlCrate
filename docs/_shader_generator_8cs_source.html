715  private static void CalcAttn(GXAttnFn fn, bool colorPass, string illumName)
716  {
717  string suffix = colorPass ? colorPassSuffix : alphaPassSuffix;
718  string part = colorPass ? ".rgb" : ".a";
719 
720  Comment("{0} attenuation pass", colorPass ? "Color" : "Alpha");
721 
722  if (fn == GXAttnFn.None)
723  {
724  Comment("No Attn");
725 
726  //Use diffuse color
727  wl("{0}{3} = {1}.{2}{3};", lightColorName, lightName, LightColorName, part);
728  }
729  else
730  {
731  //If specular, don't do anything if this isn't a specular light
732  if (fn == GXAttnFn.Specular)
733  {
734  Comment("Spec Attn");
735  wl("if ({0}.{1} != 0)", lightName, LightSpecEnabledName);
736  OpenBracket();
737  }
738  else
739  {
740  Comment("Spot Attn");
741  }
742 
743  //Write floats for light attenuation color equation
744  wl("float {0}{1} = 1.0;", AAttName, suffix); //Angular attenuation
745  wl("float {0}{1} = 1.0;", distName, suffix); //Distance attenuation
746 
747  switch (fn)
748  {
749  case GXAttnFn.Spotlight:
750 
751  wl("k0 = {0}.{1}[0];", lightName, LightDistCoefsName);
752  wl("k1 = {0}.{1}[1];", lightName, LightDistCoefsName);
753  wl("k2 = {0}.{1}[2];", lightName, LightDistCoefsName);
754  wl("a0 = {0}.{1}[0];", lightName, LightSpotCoefsName);
755  wl("a1 = {0}.{1}[1];", lightName, LightSpotCoefsName);
756  wl("a2 = {0}.{1}[2];", lightName, LightSpotCoefsName);
757 
758  //Use diffuse color
759  wl("{0}{3} = {1}.{2}{3};",
760  lightColorName, lightName, LightColorName, part);
761 
762  //Set dist
763  //length = sqrt(dot(lightVec, lightVec))
764  wl("{0}{2} = length({1});",
765  distName, lightVecName, suffix);
766 
767  //Set angular value
768  wl("{0}{3} = satlf(dot({1}, {2}));",
769  AAttName, lightVecName, lightDirName, suffix);
770 
771  break;
772  case GXAttnFn.Specular:
773 
774  wl("k0 = {0}.{1}[0];", lightName, LightDistCoefsSpecName);
775  wl("k1 = {0}.{1}[1];", lightName, LightDistCoefsSpecName);
776  wl("k2 = {0}.{1}[2];", lightName, LightDistCoefsSpecName);
777  wl("a0 = 0.0;");
778  wl("a1 = 0.0;");
779  wl("a2 = 1.0;");
780 
781  //Use specular color
782  wl("{0}{3} = {1}.{2}{3};", lightColorName, lightName, LightSpecColorName, part);
783 
784  string halfVecName = "halfVec";
785  wl("vec3 {0} = normalize({1} + normalize(-{2}));",
786  halfVecName, lightVecName, _vPositionName);
787 
788  //Set dist
789  wl("{0}{4} = {1} > 0.0 ? satlf(dot(normalize({2}), {3})): 0.0;",
790  distName, NdotLName, _vNormalName, halfVecName, suffix);
791 
792  //Set angular value, same as dist
793  wl("{0}{2} = {1}{2};",
794  AAttName, distName, suffix);
795 
796  break;
797  }
798 
799  string equation = "{2}2 * {0}{1} * {0}{1} + {2}1 * {0}{1} + {2}0";
800  string numerator = string.Format(equation, AAttName, suffix, "a");
801  string denominator = string.Format(equation, distName, suffix, "k");
802 
803  wl("{0}{3} *= satlf({1}) / ({2});", attnName, numerator, denominator, suffix);
804 
805  if (fn == GXAttnFn.Specular)
806  {
807  CloseBracket();
808  wl("else");
809  OpenBracket();
810  wl("{0}{1} *= 0.0;", attnName, suffix);
811  CloseBracket();
812  }
813  }
814 
815  //Add the light color multiplied by the attenuation
816  wl("{0}{4} += ({3}{4} * {1}{2});", illumName, attnName, suffix, lightColorName, part);
817  }

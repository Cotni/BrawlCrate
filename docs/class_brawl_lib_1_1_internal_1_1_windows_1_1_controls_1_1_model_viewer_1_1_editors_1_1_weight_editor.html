592 if (node is MDL0BoneNode)

603 int selectedIndex = weights.Select(x => x.Bone).ToArray().IndexOf(TargetBone);

608 }

612
613 //Can't do anything to a locked weight
614 if (targetWeight.Locked)

618

626 }

638 break;

658 }

669
670 //Get the change in value, divide it by all other editable weights,
671 //and then add that value to those weights to bring the overall weight sum back to 1.0f

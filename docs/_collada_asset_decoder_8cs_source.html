607  private static void RunPrimitiveCmd(byte** pIn, byte** pOut, PrimitiveDecodeCommand* pCmd, int cmdCount,
608  ushort* pIndex, int count)
609  {
610  int buffer;
611  while (count-- > 0)
612  {
613  for (int i = 0; i < cmdCount; i++)
614  {
615  buffer = pCmd[i].Index;
616  switch ((SemanticType) pCmd[i].Cmd)
617  {
618  case SemanticType.None:
619  *pIndex += 1;
620  break;
621 
622  case SemanticType.VERTEX:
623  //Can't do remap table because weights haven't been assigned yet!
624  *(ushort*) pOut[buffer] = *pIndex++;
625  pOut[buffer] += 2;
626  break;
627 
628  case SemanticType.NORMAL:
629  *(Vector3*) pOut[buffer] = ((Vector3*) pIn[buffer])[*pIndex++];
630  pOut[buffer] += 12;
631  break;
632 
633  case SemanticType.COLOR:
634  float* p = (float*) (pIn[buffer] + *pIndex++ * 16);
635  byte* p2 = pOut[buffer];
636  for (int x = 0; x < 4; x++)
637  {
638  *p2++ = (byte) (*p++ * 255.0f + 0.5f);
639  }
640 
641  pOut[buffer] = p2;
642  break;
643 
644  case SemanticType.TEXCOORD:
645  //Flip y axis so coordinates are bottom-up
646  Vector2 v = ((Vector2*) pIn[buffer])[*pIndex++];
647  v._y = 1.0f - v._y;
648  *(Vector2*) pOut[buffer] = v;
649  pOut[buffer] += 8;
650  break;
651  }
652  }

292  {
293  ResourceNode current;
294 
295  int len;
296  char* cPtr;
297  sbyte* start, end;

302  {
303  ;
304  }

307

315

317  {
318  for (cPtr = p, start = str; start < end && *start == *cPtr; start++, cPtr++)
319  {

321  }
322  }
323

422  public Color CheckCode(out List<GCTCodeLine> lines)
423  {
424  lines = new List<GCTCodeLine>();
425 
426  string code = txtCode.Text;
427  string[] values = code.Split('\n');
428  int x = 0;
429  foreach (string s in values)
430  {
431  ++x;
432  if (string.IsNullOrEmpty(s))
433  {
434  continue;
435  }
436 
437  string line = s.StartsWith("* ") ? s.Substring(2) : s;
438 
439  string[] split = line.Split(' ');
440  if (split.Length < 1)
441  {
442  continue;
443  }
444 
445  if (split[0].Length != 8)
446  {
447  return Error(x, "First value must be 8 characters long.");
448  }
449 
450  if (!uint.TryParse(split[0], NumberStyles.HexNumber, CultureInfo.InvariantCulture, out uint val1))
451  {
452  return Error(x, "First value is not a hex integer.");
453  }
454 
455  if (split.Length < 2 || string.IsNullOrWhiteSpace(split[1]))
456  {
457  return Error(x, "Needs two values.");
458  }
459 
460  if (split[1].Length != 8)
461  {
462  return Error(x, "Second value must be 8 characters long.");
463  }
464 
465  if (!uint.TryParse(split[1], NumberStyles.HexNumber, CultureInfo.InvariantCulture, out uint val2))
466  {
467  return Error(x, "Second value is not a hex integer.");
468  }
469 
470  if (split.Length > 2)
471  {
472  for (int i = 2; i < split.Length; i++)
473  {
474  if (!string.IsNullOrWhiteSpace(split[i]))
475  {
476  return Error(x, "Too many values.");
477  }
478  }
479  }
480 
481  lines.Add(new GCTCodeLine(val1, val2));
482  }
483 
484  status.Text = "Code successfully parsed";
485  return Color.Green;
486  }

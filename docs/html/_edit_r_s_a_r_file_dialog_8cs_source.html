428  public void TargetChanged()
429  {
430  dataListBox.Items.Clear();
431  soundsListBox.Items.Clear();
432 
433  if (TargetNode is RSEQNode)
434  {
435  splitter2.Visible = panel4.Visible = audioPlaybackPanel1.Visible = false;
436 
437  dataListBox.Items.AddRange(TargetNode.Children.ToArray());
438  if (dataListBox.Items.Count > 0)
439  {
440  dataListBox.SelectedIndex = 0;
441  }
442  }
443  else if (TargetNode is RBNKNode || TargetNode is RWSDNode)
444  {
445  splitter2.Visible = panel4.Visible = audioPlaybackPanel1.Visible = true;
446 
447  dataListBox.Items.AddRange(TargetNode.Children[0].Children.ToArray());
448  if (dataListBox.Items.Count > 0)
449  {
450  dataListBox.SelectedIndex = 0;
451  }
452 
453  if (TargetNode.Children.Count > 1)
454  {
455  soundsListBox.Items.AddRange(TargetNode.Children[1].Children.ToArray());
456  }
457 
458  if (soundsListBox.Items.Count > 0)
459  {
460  soundsListBox.SelectedIndex = 0;
461  }
462  }
463 
464  button1.Visible =
465  dataNewNullEntry.Visible =
466  dataNewInstParam.Visible =
467  dataNewRange.Visible =
468  dataNewIndex.Visible = TargetNode is RBNKNode;
469 
470  if (TargetNode != null)
471  {
472  Text = "Edit RSAR File - " + TargetNode.Name;
473  }
474  else
475  {
476  Text = "Edit RSAR File";
477  }
478  }

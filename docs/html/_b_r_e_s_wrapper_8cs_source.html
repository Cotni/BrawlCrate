282  public void ImportTexture()
283  {
284  if (Program.OpenFiles(FileFilters.TEX0, out string[] paths) > 0)
285  {
286  foreach (string path in paths)
287  {
288  if (path.EndsWith(".tex0", StringComparison.OrdinalIgnoreCase))
289  {
290  TEX0Node node = NodeFactory.FromFile(null, path) as TEX0Node;
291  if (node == null)
292  {
293  continue;
294  }
295 
296  ((BRRESNode) _resource).GetOrCreateFolder<TEX0Node>().AddChild(node);
297 
298  string palette = Path.ChangeExtension(path, ".plt0");
299  if (File.Exists(palette) && node.HasPalette)
300  {
301  PLT0Node n = NodeFactory.FromFile(null, palette) as PLT0Node;
302  ((BRRESNode) _resource).GetOrCreateFolder<PLT0Node>().AddChild(n);
303  }
304 
305  BaseWrapper w = FindResource(node, true);
306  w.EnsureVisible();
307  w.TreeView.SelectedNode = w;
308  }
309  else
310  {
311  using (TextureConverterDialog dlg = new TextureConverterDialog())
312  {
313  dlg.ImageSource = path;
314  if (dlg.ShowDialog(MainForm.Instance, Resource as BRRESNode) == DialogResult.OK)
315  {
316  BaseWrapper w = FindResource(dlg.TEX0TextureNode, true);
317  w.EnsureVisible();
318  w.TreeView.SelectedNode = w;
319  }
320  }
321  }
322  }

630  {
631  foreach (string resourceName in resourceNames)
632  {
633  Assembly assembly = Assembly.GetAssembly(typeof(ResourceNode));
634  using (Stream stream = assembly.GetManifestResourceStream(resourceName))
635  {

645  ((BRRESNode) _resource).GetOrCreateFolder<T>().AddChild(node);

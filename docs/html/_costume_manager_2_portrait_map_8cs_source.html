216  private static Dictionary<int, int[]> CompilePM35Mappings()
217  {
218  Dictionary<int, int[]> ret = new Dictionary<int, int[]>();
219  for (int key = 0; key <= 46; key++)
220  {
221  switch (key)
222  {
223  // Some of these characters have their portraits in a different order than Brawl, or have their additional portraits "out of order."
224  case 0:
225  // Mario
226  ret.Add(key, new int[] {0, 6, 3, 2, 5, 7, 11, 8, 9, 10, /*end*/ 1, 4});
227  break;
228  case 3:
229  // Samus
230  ret.Add(key, new int[] {0, 3, 1, 4, 2, 5, 7, 8, 6});
231  break;
232  case 12:
233  // Fox
234  ret.Add(key, new int[] {0, 5, 1, 3, 2, 4, 7, 6});
235  break;
236  case 23:
237  // Zero Suit Samus
238  ret.Add(key, new int[] {0, 3, 1, 5, 4, 2, 7, 8, 6});
239  break;
240  case 27:
241  // Mewtwo - uses Pokémon Trainer's character index
242  ret.Add(key, new int[] {0, 2, 1, 3, 4, 5});
243  break;
244  case 28:
245  // Charizard
246  ret.Add(key, new int[] {0, 1, 3, 2, 4, 5, 6});
247  break;
248  case 29:
249  // Squirtle
250  ret.Add(key, new int[] {0, 2, 1, 3, 4, 5, 6, 7});
251  break;
252  case 37:
253  // Wario
254  ret.Add(key, new int[] {6, 7, 10, 8, 11, 9, 0, 1, 3, 2, 5, 4});
255  break;
256  case 39:
257  // Roy
258  ret.Add(key, new int[] {0, 2, 1, 3, 4, 5, 6});
259  break;
260  case 46:
261  // Sonic
262  ret.Add(key, new int[] {0, 1, 4, 2, 5, 6, 7, 8});
263  break;
264  default:
265  if (!PortraitToCostumeMappings.ContainsKey(key))
266  {
267  continue;
268  }
269 
270  /* All other characters in PM 3.6 follow a pattern: the portraits start out in the same order Brawl has them,
271  and any additional portraits are in order after the highest-numbered original portrait. */
272  int[] arr1 = PortraitToCostumeMappings[key];
273  int max = arr1.Max();
274  int[] arr2 = new int[12];
275  Array.Copy(arr1, arr2, arr1.Length);
276  for (int i = arr1.Length; i < 12; i++)
277  {
278  arr2[i] = ++max;
279  }
280 
281  ret.Add(key, arr2);
282  break;
283  }
284  }
285 
286  return ret;

526  public static string[] ForPac(TracklistModifier modifier, CMM cmm, string filename)
527  {
528  filename = filename.ToLower();
529  string key = filename.Substring(0, filename.IndexOfAny("_.".ToCharArray()));
530  int stageId = StageIDMap.StageIDForPac(filename);
531  if (cmm != null && cmm.Map.TryGetValue(stageId, out IEnumerable<Song> songs))
532  {
533  return songs.Select(s => s.Filename).ToArray();
534  }
535 
536  // Tracklist Modifier 1.0 [Phantom Wings]
537  if (cmm.TracklistModifier != null)
538  {
539  byte currentTracklistId = cmm.TracklistModifier[stageId];
540  for (int i = 0; i < CMM.StandardCMMTracklistModifierData.Length; i++)
541  {
542  if (CMM.StandardCMMTracklistModifierData[i] == currentTracklistId)
543  {
544  int originalStageId = i;
545  // Make sure the index is a valid stage ID. Some bytes
546  // appear more than once, the first time at an index
547  // that's not a valid stage ID (e.g. Battlefield's 00
548  // at index 0 or Bridge of Eldin's 0A at index 10.)
549  if (Stage.Stages.Any(s => s.ID == i))
550  {
551  stageId = originalStageId;
552  break;
553  }
554  }
555  }
556  }
557 
558  // Tracklist Modifier [standardtoaster]
559  stageId = modifier[stageId];
560  if (stageId == 5 && filename.StartsWith("stgmariopast_00"))
561  {
562  return new[]
563  {
564  "A01",
565  "A16",
566  "A10"
567  };
568  }
569  else if (stageId == 5 && filename.StartsWith("stgmariopast_01"))
570  {
571  return new[]
572  {
573  "A02",
574  "A03",
575  "A04"
576  };
577  }
578  else if (dict.TryGetValue(stageId, out string[] ret))

178  {

180  int[] p;

182  int previouscol, startpos;
183 
184  previouscol = 0;
185  startpos = 0;
186  for (i = 0; i < netsize; i++)
187  {
188  p = network[i];
189  smallpos = i;
190  smallval = p[1]; /* index on g */
191  /* find smallest in i..netsize-1 */
192  for (j = i + 1; j < netsize; j++)
193  {
194  q = network[j];
195  if (q[1] < smallval)
196  {
197  /* index on g */
198  smallpos = j;
199  smallval = q[1]; /* index on g */
200  }
201  }
202 
203  q = network[smallpos];
204  /* swap p (i) and q (smallpos) entries */
205  if (i != smallpos)
206  {
207  j = q[0];
208  q[0] = p[0];
209  p[0] = j;
210  j = q[1];
211  q[1] = p[1];
212  p[1] = j;
213  j = q[2];
214  q[2] = p[2];
215  p[2] = j;
216  j = q[3];
217  q[3] = p[3];
218  p[3] = j;
219  }
220

222  if (smallval != previouscol)
223  {
224  netindex[previouscol] = (startpos + i) >> 1;
225  for (j = previouscol + 1; j < smallval; j++)
226  {

228  }
229

232  }
233  }
234

236  for (j = previouscol + 1; j < 256; j++)

244  public unsafe void Learn()
245  {
246  int i, j, b, g, r;
247  int radius, rad, alpha, step, delta, samplepixels;
248  UnsafeBuffer p;
249  int pix, lim;
250 
251  if (lengthcount < minpicturebytes)
252  {
253  samplefac = 1;
254  }
255 
256  alphadec = 30 + (samplefac - 1) / 3;
257  p = thepicture;
258  pix = 0;
259  lim = lengthcount;
260  samplepixels = lengthcount / (3 * samplefac);
261  delta = samplepixels / ncycles;
262  alpha = initalpha;
263  radius = initradius;
264 
265  rad = radius >> radiusbiasshift;
266  if (rad <= 1)
267  {
268  rad = 0;
269  }
270 
271  for (i = 0; i < rad; i++)
272  {
273  radpower[i] =
274  alpha * ((rad * rad - i * i) * radbias / (rad * rad));
275  }
276 
277  //fprintf(stderr,"beginning 1D learning: initial radius=%d\n", rad);
278 
279  if (lengthcount < minpicturebytes)
280  {
281  step = 3;
282  }
283  else if (lengthcount % prime1 != 0)
284  {
285  step = 3 * prime1;
286  }
287  else
288  {
289  if (lengthcount % prime2 != 0)
290  {
291  step = 3 * prime2;
292  }
293  else
294  {
295  if (lengthcount % prime3 != 0)
296  {
297  step = 3 * prime3;
298  }
299  else
300  {
301  step = 3 * prime4;
302  }
303  }
304  }
305 
306  i = 0;
307  byte* addr = (byte*) p.Address;
308  while (i < samplepixels)
309  {
310  b = (addr[pix + 0] & 0xff) << netbiasshift;
311  g = (addr[pix + 1] & 0xff) << netbiasshift;
312  r = (addr[pix + 2] & 0xff) << netbiasshift;
313  j = Contest(b, g, r);
314 
315  Altersingle(alpha, j, b, g, r);
316  if (rad != 0)
317  {
318  Alterneigh(rad, j, b, g, r); /* alter neighbours */
319  }
320 
321  pix += step;
322  if (pix >= lim)
323  {
324  pix -= lengthcount;
325  }
326 
327  i++;
328  if (delta == 0)
329  {
330  delta = 1;
331  }
332 
333  if (i % delta == 0)
334  {
335  alpha -= alpha / alphadec;
336  radius -= radius / radiusdec;
337  rad = radius >> radiusbiasshift;
338  if (rad <= 1)
339  {
340  rad = 0;
341  }
342 
343  for (j = 0; j < rad; j++)
344  {
345  radpower[j] =
346  alpha * ((rad * rad - j * j) * radbias / (rad * rad));
347  }
348  }
349  }
350 
351  //fprintf(stderr,"finished 1D learning: readonly alpha=%f !\n",((float)alpha)/initalpha);
352  }

477  --------------------------------------------------------------------------------- */
478  protected void Alterneigh(int rad, int i, int b, int g, int r)
479  {
480  int j, k, lo, hi, a, m;
481  int[] p;
482 
483  lo = i - rad;
484  if (lo < -1)
485  {
486  lo = -1;
487  }
488 
489  hi = i + rad;
490  if (hi > netsize)
491  {
492  hi = netsize;
493  }
494 
495  j = i + 1;
496  k = i - 1;
497  m = 1;
498  while (j < hi || k > lo)
499  {
500  a = radpower[m++];
501  if (j < hi)
502  {
503  p = network[j++];
504  try
505  {

507  p[1] -= a * (p[1] - g) / alpharadbias;
508  p[2] -= a * (p[2] - r) / alpharadbias;
509  }
510  catch (Exception)
511  {
512  // prevents 1.3 miscompilation
513  }
514  }
515 
516  if (k > lo)
517  {
518  p = network[k--];
519  try
520  {
521  p[0] -= a * (p[0] - b) / alpharadbias;
522  p[1] -= a * (p[1] - g) / alpharadbias;
523  p[2] -= a * (p[2] - r) / alpharadbias;

533  /* Move neuron i towards biased (b,g,r) by factor alpha
534  ---------------------------------------------------- */
535  protected void Altersingle(int alpha, int i, int b, int g, int r)
536  {
537  /* alter hit neuron */
538  int[] n = network[i];
539  n[0] -= alpha * (n[0] - b) / initalpha;
540  n[1] -= alpha * (n[1] - g) / initalpha;
541  n[2] -= alpha * (n[2] - r) / initalpha;

544  /* Search for biased BGR values
545  ---------------------------- */
546  protected int Contest(int b, int g, int r)
547  {
548  /* finds closest neuron (min dist) and updates freq */
549  /* finds best neuron (min dist-bias) and returns position */
550  /* for frequently chosen neurons, freq[i] is high and bias[i] is negative */
551  /* bias[i] = gamma*((1/netsize)-freq[i]) */
552 
553  int i, dist, a, biasdist, betafreq;
554  int bestpos, bestbiaspos, bestd, bestbiasd;
555  int[] n;
556 
557  bestd = ~(1 << 31);
558  bestbiasd = bestd;
559  bestpos = -1;
560  bestbiaspos = bestpos;
561 
562  for (i = 0; i < netsize; i++)
563  {
564  n = network[i];
565  dist = n[0] - b;
566  if (dist < 0)
567  {
568  dist = -dist;
569  }
570 
571  a = n[1] - g;
572  if (a < 0)
573  {
574  a = -a;
575  }
576 
577  dist += a;
578  a = n[2] - r;
579  if (a < 0)
580  {
581  a = -a;
582  }
583 
584  dist += a;
585  if (dist < bestd)
586  {
587  bestd = dist;
588  bestpos = i;
589  }
590 
591  biasdist = dist - (bias[i] >> (intbiasshift - netbiasshift));
592  if (biasdist < bestbiasd)
593  {
594  bestbiasd = biasdist;
595  bestbiaspos = i;
596  }
597 
598  betafreq = freq[i] >> betashift;
599  freq[i] -= betafreq;
600  bias[i] += betafreq << gammashift;
601  }
602 
603  freq[bestpos] += beta;
604  bias[bestpos] -= betagamma;
605  return bestbiaspos;
606  }

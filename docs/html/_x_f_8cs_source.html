68  public static unsafe List<XFData> Parse(VoidPtr address)
69  {
70  List<XFData> XFCmds = new List<XFData>();
71  byte* pData = (byte*) address;
72  Top:
73  if (*pData++ == 0x10)
74  {
75  XFData dat = new XFData();
76  int count = *(bushort*) pData;
77  pData += 2;
78  dat._addr = (XFMemoryAddr) (ushort) *(bushort*) pData;
79  pData += 2;
80  dat._values = new List<uint>();
81  for (int i = 0; i < count + 1; i++)
82  {
83  dat._values.Add(*(buint*) pData);
84  pData += 4;
85  }
86 
87  XFCmds.Add(dat);
88  goto Top;
89  }
90 
91  return XFCmds;
92  }

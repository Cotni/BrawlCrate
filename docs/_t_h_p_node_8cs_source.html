208  public Bitmap GetImage()
209  {
210  //We have to convert the raw buffer to a usable image every time the image is called.
211  //Dispose of image when done displaying. This way we won't run out of memory.
212  //Doesn't seem to slow down frame rate or anything.
213 
214  byte[] buffer = new byte[Header->CompAddr[0]];
215  Marshal.Copy(Header->GetComp(_node._componentTypes.Count, 0), buffer, 0, buffer.Length);
216 
217  bool begun = false;
218  List<byte> temp = buffer.ToList();
219 
220  int end = 0;
221  for (int i = temp.Count - 2; i >= 0; i--)
222  {
223  byte b1 = temp[i];
224  byte b2 = temp[i + 1];
225  ushort code = (ushort) ((b1 << 8) | b2);
226 
227  if (Enum.IsDefined(typeof(JpegMarkers), code))
228  {
229  JpegMarkers m = (JpegMarkers) code;
230  if (m == JpegMarkers.EndOfImage)
231  {
232  end = i;
233  break;
234  }
235  }
236  }
237 
238  for (int i = 0; i < temp.Count; i++)
239  {
240  byte b1 = temp[i];
241  if (b1 == 0xFF)
242  {
243  byte b2 = temp[i + 1];
244  ushort code = (ushort) ((b1 << 8) | b2);
245 
246  if (Enum.IsDefined(typeof(JpegMarkers), code))
247  {
248  JpegMarkers m = (JpegMarkers) code;
249  if (m == JpegMarkers.EndOfImage && i == end)
250  {
251  break;
252  }
253  else
254  {
255  if (begun)
256  {
257  temp.Insert(i + 1, 0);
258  i++;
259  }
260  }
261 
262  if (m == JpegMarkers.StartOfScan)
263  {
264  begun = true;
265  }
266  }
267  else
268  {
269  if (begun)
270  {
271  temp.Insert(i + 1, 0);

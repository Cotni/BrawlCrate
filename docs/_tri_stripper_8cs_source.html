139  {
140  m_TriHeap.Reserve(m_Triangles.Count);
141 
142  //Set up the triangles priority queue
143  //The lower the number of available neighbour triangles, the higher the priority.
144  for (uint i = 0; i < m_Triangles.Count; i++)
145  {
146  m_TriHeap.Push(m_Triangles[i].Size);
147  }

160  private void Stripify()
161  {
162  while (!m_TriHeap.Empty)
163  {
164  //There is no triangle in the candidates list, refill it with the loneliest triangle
165  uint HeapTop = m_TriHeap.Position(0);
166  m_Candidates.Add(HeapTop);
167 
168  while (m_Candidates.Count != 0)
169  {
170  //Note: FindBestStrip empties the candidate list, while BuildStrip refills it
171  Strip TriStrip = FindBestStrip();
172 
173  if (TriStrip.Size >= m_MinStripSize)
174  {
175  BuildStrip(TriStrip);
176  }
177  }
178 
179  if (!m_TriHeap.Removed(HeapTop))
180  {
181  m_TriHeap.Erase(HeapTop);
182  }
183 
184  //Eliminate all the triangles that have now become useless
185  while (!m_TriHeap.Empty && m_TriHeap.Top == 0)
186  {
187  m_TriHeap.Pop();

221  private Strip FindBestStrip()
222  {
223  //Allow to restore the cache (modified by ExtendTriToStrip) and implicitly reset the cache hit count
224  CacheSimulator CacheBackup = m_Cache;
225 
226  Policy policy = new Policy(m_MinStripSize, Cache);
227 
228  while (m_Candidates.Count != 0)
229  {
230  uint Candidate = m_Candidates[m_Candidates.Count - 1];
231  m_Candidates.RemoveAt(m_Candidates.Count - 1);
232 
233  //Discard useless triangles from the candidate list
234  if (m_Triangles[Candidate].Marked || m_TriHeap[Candidate] == 0)
235  {
236  continue;
237  }
238 
239  //Try to extend the triangle in the 3 possible forward directions
240  for (uint i = 0; i < 3; i++)
241  {
242  Strip Strip = ExtendToStrip(Candidate, (TriOrder) i);
243  policy.Challenge(Strip, m_TriHeap[Strip.Start], m_Cache.HitCount);
244 
245  m_Cache = CacheBackup;
246  }
247 
248  //Try to extend the triangle in the 6 possible backward directions
249  if (m_BackwardSearch)
250  {
251  for (uint i = 0; i < 3; i++)
252  {
253  Strip Strip = BackExtendToStrip(Candidate, (TriOrder) i, false);
254  if (Strip != null)
255  {
256  policy.Challenge(Strip, m_TriHeap[Strip.Start], m_Cache.HitCount);
257  }
258 
259  m_Cache = CacheBackup;
260  }
261 
262  for (uint i = 0; i < 3; i++)
263  {
264  Strip Strip = BackExtendToStrip(Candidate, (TriOrder) i, true);
265  if (Strip != null)
266  {
267  policy.Challenge(Strip, m_TriHeap[Strip.Start], m_Cache.HitCount);
268  }
269 
270  m_Cache = CacheBackup;
271  }
272  }
273  }

324  private Strip BackExtendToStrip(uint Start, TriOrder Order, bool ClockWise)
325  {
326  m_CurrentNodes = new List<ushort>();
327  _checkNodes = true;
328 
329  //Begin a new strip
330  Triangle tri = m_Triangles[Start].m_Elem;
331  uint b = LastEdge(tri, Order).B;
332  if (TryAddNode(b))

336  }

345  GraphArray<Triangle>.Node Node = null;
346 
347  //Loop while we can further extend the strip
348  for (uint i = Start; !Cache || Size + 2 < CacheSize; Size++)
349  {

351  GraphArray<Triangle>.Arc Link = BackLinkToNeighbour(Node, ClockWise, ref Order);
352

411  private GraphArray<Triangle>.Arc LinkToNeighbour(GraphArray<Triangle>.Node Node, bool ClockWise,
412  ref TriOrder Order, bool NotSimulation)
413  {
414  TriangleEdge Edge = LastEdge(Node.m_Elem, Order);
415  for (uint i = Node.m_Begin; i < Node.m_End; i++)
416  {
417  GraphArray<Triangle>.Arc Link = Node.Arcs[(int) i];
418 
419  //Get the reference to the possible next triangle
420  Triangle Tri = Link.Terminal.m_Elem;
421 
422  //Check whether it's already been used
423  if ((NotSimulation || Tri.StripID != m_StripID) && !Link.Terminal.Marked)
424  {
425  //Does the current candidate triangle match the required position for the strip?
426  if (Edge.B == Tri.A && Edge.A == Tri.B && TryAddNode(Tri.C))
427  {
428  Order = ClockWise ? TriOrder.ABC : TriOrder.BCA;
429  AddIndex(Tri.C, NotSimulation);
430  return Link;
431  }
432  else if (Edge.B == Tri.B && Edge.A == Tri.C && TryAddNode(Tri.A))
433  {
434  Order = ClockWise ? TriOrder.BCA : TriOrder.CAB;
435  AddIndex(Tri.A, NotSimulation);
436  return Link;
437  }
438  else if (Edge.B == Tri.C && Edge.A == Tri.A && TryAddNode(Tri.B))
439  {
440  Order = ClockWise ? TriOrder.CAB : TriOrder.ABC;
441  AddIndex(Tri.B, NotSimulation);
442  return Link;
443  }
444  }
445  }
446 
447  return null;

696  private static int BinarySearch<T>(IList<T> list, T value, Comparison<T> comp)
697  {
698  int lo = 0, hi = list.Count - 1;
699  while (lo < hi)
700  {
701  int m = (hi + lo) / 2;
702  if (comp(list[m], value) < 0)
703  {
704  lo = m + 1;
705  }
706  else
707  {
708  hi = m - 1;
709  }
710  }

713  {
714  lo++;
715  }
